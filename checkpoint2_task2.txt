V1:[4,-5,8];
V2:[2,6,3];
PS:FLOAT;
BEGIN
PS:dot_product(V1,V2);
IF (PS==0) THEN DO
 Write("the vectors are orthogonal");
END_IF
END

FUNCTION dot_product(v1,v2:ARRAY_OF FLOAT):FLOAT

VAR 
i: INTEGER;
ps:=0:FLOAT;
n:=v1.length:INTEGER;
BEGIN
FOR i FROM 0 TO n-1 DO
 ps= ps + v1[i]*v2[i];
END_FOR
RETURN ps;
END
